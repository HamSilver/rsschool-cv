function sumIntervals(intervals){
	if (intervals.length>1) {
		intervals.sort((c,n) => c[0]- n[0]);
		let st = [intervals[0]];
		for (let i = 1; i < intervals.length; i++) {
			if (intervals[i][0]<=st[st.length-1][1]) {
				st.splice(
					st.length-1,
					1,
					[Math.min(st[st.length-1][0],intervals[i][0]),
					Math.max(st[st.length-1][1],intervals[i][1])]
				);
			} else {
				st.push(intervals[i]);
			}
		}
		intervals=st;
	}
	let r=0;
	for(let e of intervals) {
		r+=e[1]-e[0];
	}
	return r;
}
